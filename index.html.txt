            <option value="ru">Ruscha</option>

        <select id="targetLang">
            <option value="uz">O'zbekcha</option>

            <option value="ru">Ruscha</option>
            <!-- Boshqa tillarni qo'shish mumkin -->
        </select>

        <button onclick="translateText()">Tarjima qilish</button>

        <p id="result" style="margin-top: 20px; font-weight: bold;"></p>

    <script>
        function translateText() {
            const inputText = document.getElementById('inputText').value;
            const sourceLang = document.getElementById('sourceLang').value;
            const targetLang = document.getElementById('targetLang').value;
            const resultElement = document.getElementById('result');

            if (inputText.trim() === "") {
                resultElement.textContent = "Iltimos, matn kiriting!";
                return;
            }

            const apiUrl = `https://api.mymemory.translated.net/get?q=${encodeURIComponent(inputText)}&langpair=${sourceLang}|${targetLang}`;

            fetch(apiUrl)
                .then(response => response.json())
                .then(data => {
                    resultElement.textContent = "Tarjima: " + data.responseData.translatedText;
                })
                .catch(error => {
                    resultElement.textContent = "Xatolik yuz berdi, iltimos keyinroq qayta urinib ko'ring.";
                });
        }
    </script>
